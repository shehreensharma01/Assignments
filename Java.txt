package assignment.one;

public class Shape {
    double length, breadth, radius;

    // Default constructor
    Shape() {
        System.out.println("Inside Default Constructor");
        length = 0;
        breadth = 0;
        radius = 0;
    }

    // Constructor for Square
    Shape(double side) {
        length = side;
        breadth = side;
    }

    // Constructor for Rectangle
    Shape(double l, double b) {
        length = l;
        breadth = b;
    }

    // Constructor for Circle
    Shape(double r, boolean isCircle) {
        radius = r;
    }

    // Method to display area of Square
    public void displaySquare() {
        System.out.println("Area of Square = " + (length * breadth));
    }

    // Method to display area of Rectangle
    public void displayRectangle() {
        System.out.println("Area of Rectangle = " + (length * breadth));
    }

    // Method to display area of Circle
    public void displayCircle() {
        System.out.println("Area of Circle = " + (3.14 * radius * radius));
    }

    // Method overloading for Triangle
    public void area(double base, double height) {
        System.out.println("Area of Triangle = " + (0.5 * base * height));
    }

    // Method overloading for Rhombus
    public void area(double d1, double d2, String type) {
        if (type.equalsIgnoreCase("rhombus")) {
            System.out.println("Area of Rhombus = " + (0.5 * d1 * d2));
        }
    }

    public static void main(String args[]) {
        Shape s1 = new Shape();           // default constructor
        Shape s2 = new Shape(5);          // square
        Shape s3 = new Shape(4, 6);       // rectangle
        Shape s4 = new Shape(7, true);    // circle

        s2.displaySquare();
        s3.displayRectangle();
        s4.displayCircle();

        s1.area(10, 12);          // triangle
        s1.area(6, 8, "rhombus"); // rhombus
    }
}

package assignment.two;

public class Student {
    String name;
    int age;
    char section;
    char gender;
    int subject1, subject2, subject3;
    int total;
    double percentage;

    // Constructor with 3 subjects
    Student(String n, int a, char sec, char g, int s1, int s2, int s3) {
        name = n;
        age = a;
        section = sec;
        gender = g;
        subject1 = s1;
        subject2 = s2;
        subject3 = s3;
        calculate();
    }

    // Constructor with only 2 subjects (subject1 = 0)
    Student(String n, int a, char sec, char g, int s2, int s3) {
        name = n;
        age = a;
        section = sec;
        gender = g;
        subject1 = 0;
        subject2 = s2;
        subject3 = s3;
        calculate();
    }

    // Method to calculate total and percentage
    void calculate() {
        total = subject1 + subject2 + subject3;
        percentage = total / 3.0;
    }

    // Display method
    void display() {
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
        System.out.println("Section: " + section);
        System.out.println("Gender: " + gender);
        System.out.println("Marks: " + subject1 + ", " + subject2 + ", " + subject3);
        System.out.println("Total: " + total);
        System.out.println("Percentage: " + percentage + "%");
        System.out.println("-----------------------------");
    }

    public static void main(String args[]) {
        Student s1 = new Student("Armaan", 20, 'A', 'M', 80, 75, 90);
        Student s2 = new Student("Bharat", 19, 'B', 'F', 85, 88);
        Student s3 = new Student("Rahat", 21, 'A', 'M', 70, 65);
        Student s4 = new Student("Rahul", 20, 'C', 'F', 95, 90, 92);

        s1.display();
        s2.display();
        s3.display();
        s4.display();
    }
}
